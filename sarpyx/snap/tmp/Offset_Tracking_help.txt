OPERATOR: Offset-Tracking
----------------------------------------
Usage:
  gpt Offset-Tracking [options] 

Description:
  Create velocity vectors from offset tracking


Source Options:
  -SsourceProduct=<file>    Sets source 'sourceProduct' to <filepath>.
                            This is a mandatory source.

Parameter Options:
  -PaverageBoxSize=<string>              Sets parameter 'averageBoxSize' to <string>.
                                         Value must be one of '3', '5', '9', '11'.
                                         Default value is '5'.
  -PfillHoles=<boolean>                  Sets parameter 'fillHoles' to <boolean>.
                                         Default value is 'true'.
  -PgridAzimuthSpacing=<int>             The output grid azimuth spacing in pixels
                                         Valid interval is (1, *).
                                         Default value is '40'.
  -PgridRangeSpacing=<int>               The output grid range spacing in pixels
                                         Valid interval is (1, *).
                                         Default value is '40'.
  -PmaxVelocity=<double>                 The threshold for eliminating invalid GCPs
                                         Valid interval is (0, *).
                                         Default value is '5.0'.
  -Pradius=<int>                         Radius for Hole-Filling
                                         Valid interval is (0, *).
                                         Default value is '4'.
  -PregistrationOversampling=<string>    Sets parameter 'registrationOversampling' to <string>.
                                         Value must be one of '2', '4', '8', '16', '32', '64', '128', '256', '512'.
                                         Default value is '16'.
  -PregistrationWindowHeight=<string>    Sets parameter 'registrationWindowHeight' to <string>.
                                         Value must be one of '32', '64', '128', '256', '512', '1024', '2048'.
                                         Default value is '128'.
  -PregistrationWindowWidth=<string>     Sets parameter 'registrationWindowWidth' to <string>.
                                         Value must be one of '32', '64', '128', '256', '512', '1024', '2048'.
                                         Default value is '128'.
  -PresamplingType=<string>              Methods for velocity interpolation.
                                         Value must be one of 'NEAREST_NEIGHBOUR', 'BILINEAR_INTERPOLATION', 'BICUBIC_INTERPOLATION', 'BISINC_5_POINT_INTERPOLATION', 'CUBIC_CONVOLUTION'.
                                         Default value is 'BICUBIC_INTERPOLATION'.
  -ProiVector=<string>                   Sets parameter 'roiVector' to <string>.
  -PspatialAverage=<boolean>             Sets parameter 'spatialAverage' to <boolean>.
                                         Default value is 'true'.
  -PxCorrThreshold=<double>              The cross-correlation threshold
                                         Valid interval is (0, *).
                                         Default value is '0.1'.

        <xCorrThreshold>double</xCorrThreshold>
        <registrationOversampling>string</registrationOversampling>
        <averageBoxSize>string</averageBoxSize>
        <maxVelocity>double</maxVelocity>

        <resamplingType>string</resamplingType>
        <spatialAverage>boolean</spatialAverage>

        <roiVector>string</roiVector>
      </parameters>

  </graph>
